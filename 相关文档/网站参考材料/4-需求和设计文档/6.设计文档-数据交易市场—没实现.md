# 数据交易市场 设计文档

## 1. 项目概述

### 1.1 设计目标

基于Jeecg Boot模板和Ant Design Vue 1.7.2，构建数据交易市场前端页面，风格与现有数据空间资产管理保持一致。

### 1.2 技术栈

| 技术             | 版本          | 说明         |
| -------------- | ----------- | ---------- |
| Vue            | 2.6.10      | 前端框架       |
| Ant Design Vue | 1.7.2       | UI组件库      |
| Vue Router     | 3.0.1       | 路由（Hash模式） |
| Axios          | 0.18.0      | HTTP请求     |
| Jeecg Boot     | 3.4.3-beta2 | 已有框架       |

### 1.3 主题配置

| 配置项   | 值         |
| ----- | --------- |
| 主色    | `#009688` |
| 侧边栏主题 | `dark`    |
| 布局模式  | `topmenu` |

#### 3.1.2 脚本逻辑

```javascript
<script>
import { getAction } from '@/api/manage'
import { JeecgListMixin } from '@/mixins/JeecgListMixin'

export default {
  name: 'MarketHome',
  mixins: [JeecgListMixin],
  data() {
    return {
      // 路由
      url: {
        list: '/marketAsset/page',
        listByIds: '/marketAsset/batchQuery'
      },
      // 查询参数
      queryParam: {
        assetType: '',
        securityLevel: '',
        keyword: '',
        // 数据集专属
        dataGranularity: '',
        // API专属
        openLevel: '',
        operationType: '',
        apiGranularity: '',
        // 训练集专属
        dataFormat: '',
        dataModality: '',
        annotationType: '',
        trainingPhase: '',
        dataSource: '',
        dataVolumeMin: '',
        dataVolumeMax: ''
      },
      // 已选筛选标签
      selectedTags: [],
      // 选中资产ID列表
      selectedIds: [],
      // 资产类型映射
      assetTypeMap: {
        dataset: { label: '数据集', color: 'blue' },
        api: { label: '数据API', color: 'purple' },
        training_set: { label: '高质量训练集', color: 'orange' }
      },
      // 数据密级映射
      securityLevelMap: {
        public: { label: '公开', color: 'green' },
        internal: { label: '内部', color: 'blue' },
        secret: { label: '秘密', color: 'orange' },
        confidential: { label: '机密', color: 'red' },
        'top-secret': { label: '绝密', color: 'purple' }
      }
    }
  },
  computed: {
    hasSelectedTags() {
      return this.selectedTags.length > 0
    }
  },
  methods: {
    // 加载数据
    loadData() {
      if (this.loading) return
      this.loading = true
      getAction(this.url.list, { ...this.queryParam, ...this.ipagination })
        .then(res => {
          if (res.success) {
            this.dataSource = res.result.records || []
            this.ipagination.total = res.result.total
          } else {
            this.$message.warning(res.message)
          }
        })
        .finally(() => {
          this.loading = false
        })
    },
    // 搜索
    handleSearch() {
      this.selectedIds = []
      this.loadData()
    },
    // 卡片点击
    handleCardClick(item) {
      this.$router.push({
        path: '/MarketAssetDetail',
        query: { id: item.id }
      })
    },
    // 选择资产
    handleSelect(item) {
      const index = this.selectedIds.indexOf(item.id)
      if (index > -1) {
        this.selectedIds.splice(index, 1)
      } else {
        this.selectedIds.push(item.id)
      }
    },
    // 分页变化
    handlePageChange(page, pageSize) {
      this.ipagination.current = page
      this.loadData()
    },
    // 获取密级颜色
    getSecurityColor(level) {
      return this.securityLevelMap[level]?.color || 'default'
    },
    // 获取密级标签
    getSecurityLabel(level) {
      return this.securityLevelMap[level]?.label || level
    },
    // 获取资产类型颜色
    getAssetTypeColor(type) {
      return this.assetTypeMap[type]?.color || 'default'
    },
    // 获取资产类型标签
    getAssetTypeLabel(type) {
      return this.assetTypeMap[type]?.label || type
    },
    // 移除筛选标签
    removeTag(key) {
      this.queryParam[key] = ''
      this.loadData()
    },
    // 清除全部标签
    clearAllTags() {
      Object.keys(this.queryParam).forEach(key => {
        if (key !== 'keyword') {
          this.queryParam[key] = ''
        }
      })
      this.queryParam.keyword = ''
      this.loadData()
    }
  }
}
</script>
```

```javascript
<script>
import { getAction, putAction } from '@/api/manage'
import { JeecgListMixin } from '@/mixins/JeecgListMixin'

export default {
  name: 'MyAssets',
  mixins: [JeecgListMixin],
  data() {
    return {
      url: {
        list: '/marketAsset/myAssets',
        delist: '/marketAsset/delist',
        relist: '/marketAsset/list'
      },
      columns: [
        { title: '资产名称', dataIndex: 'assetName', scopedSlots: { customRender: 'assetName' } },
        { title: '资产类型', dataIndex: 'assetType' },
        { title: '数据密级', dataIndex: 'securityLevel', scopedSlots: { customRender: 'securityLevel' } },
        { title: '上架时间', dataIndex: 'listedTime' },
        { title: '状态', dataIndex: 'status', scopedSlots: { customRender: 'status' } },
        { title: '操作', key: 'action', scopedSlots: { customRender: 'action' } }
      ],
      selectedIds: []
    }
  },
  methods: {
    // 加载数据
    loadData() {
      getAction(this.url.list, this.queryParam).then(res => {
        if (res.success) {
          this.dataSource = res.result.records || []
          this.pagination.total = res.result.total
        }
      })
    },
    // 下架
    handleDelist(record) {
      this.$confirm({
        title: '确认下架',
        content: `确定要下架资产「${record.assetName}」吗？下架后可重新上架。`,
        okText: '确认',
        cancelText: '取消',
        onOk: () => {
          return putAction(this.url.delist, { id: record.id }).then(res => {
            if (res.success) {
              this.$message.success('下架成功')
              this.loadData()
            } else {
              this.$message.warning(res.message)
            }
          })
        }
      })
    },
    // 批量下架
    handleBatchDelist() {
      this.$confirm({
        title: '确认批量下架',
        content: `确定要下架选中的 ${this.selectedIds.length} 项资产吗？`,
        onOk: () => {
          return putAction(this.url.delist + 'Batch', { ids: this.selectedIds }).then(res => {
            if (res.success) {
              this.$message.success('批量下架成功')
              this.selectedIds = []
              this.loadData()
            }
          })
        }
      })
    },
    // 重新上架
    handleRelist(record) {
      putAction(this.url.relist, { id: record.id }).then(res => {
        if (res.success) {
          this.$message.success('上架成功')
          this.loadData()
        }
      })
    },
    // 查看详情
    handleViewDetail(record) {
      this.$router.push({
        path: '/MarketAssetDetail',
        query: { id: record.id }
      })
    },
    // 选择变化
    onSelectChange(selectedRowKeys) {
      this.selectedIds = selectedRowKeys
    },
    getSecurityColor(level) {
      const map = { public: 'green', internal: 'blue', secret: 'orange', confidential: 'red', 'top-secret': 'purple' }
      return map[level] || 'default'
    },
    getSecurityLabel(level) {
      const map = { public: '公开', internal: '内部', secret: '秘密', confidential: '机密', 'top-secret': '绝密' }
      return map[level] || level
    }
  }
}
</script>
```

```javascript
import { getAction, postAction, putAction, deleteAction } from '@/api/manage'

/**
 * 分页查询市场资产
 * @param {Object} params - 查询参数
 * @returns {Promise}
 */
export function getMarketAssetPage(params) {
  return getAction('/marketAsset/page', params)
}

/**
 * 查询市场资产详情
 * @param {String} id - 资产ID
 * @returns {Promise}
 */
export function getMarketAssetById(id) {
  return getAction('/marketAsset/queryById', { id })
}

/**
 * 资产上架
 * @param {String|Array} assetId - 资产ID或ID数组
 * @returns {Promise}
 */
export function addToMarket(assetId) {
  const data = Array.isArray(assetId) ? { ids: assetId } : { id: assetId }
  return postAction('/marketAsset/list', data)
}

/**
 * 资产下架
 * @param {String|Array} assetId - 资产ID或ID数组
 * @returns {Promise}
 */
export function removeFromMarket(assetId) {
  const data = Array.isArray(assetId) ? { ids: assetId } : { id: assetId }
  return putAction('/marketAsset/delist', data)
}

/**
 * 查询我的资产（卖方）
 * @param {Object} params - 查询参数
 * @returns {Promise}
 */
export function getMyMarketAssets(params) {
  return getAction('/marketAsset/myAssets', params)
}

/**
 * 查询申请记录
 * @param {Object} params - 查询参数
 * @returns {Promise}
 */
export function getApplyRecords(params) {
  return getAction('/marketAsset/applyRecords', params)
}

/**
 * 审批申请
 * @param {String} id - 申请记录ID
 * @param {Boolean} approved - 是否通过
 * @returns {Promise}
 */
export function approveApply(id, approved) {
  return putAction('/marketAsset/approveApply', { id, approved })
}

/**
 * 批量查询资产（根据ID列表）
 * @param {Array} ids - 资产ID数组
 * @returns {Promise}
 */
```
